@startuml

class Groupe {
  - formation : Formation
  - etudiants : List<Etudiant>
}

class Etudiant {
  - identite : Identite
  - formation : Formation
  - resultats : Map<Matiere, List<Double>>
}

class Formation {
  - idFormation : String
  - matieres : Map<Matiere, Double>
}

class Identite {
  - nip : String
  - nom : String
  - prenom : String
}

class Matiere {
  - nom : String
}

Etudiant *-- Identite        
Formation *-- Matiere       
Etudiant --> Formation       
Groupe --> Formation        
Groupe --> Etudiant          

@enduml
