Diagramme de sequence Moyenne par matière : 

@startuml
actor "Utilisateur" as User
participant "Etudiant" as Etudiant
participant "Formation" as Formation

User -> Etudiant : calculerMoyenneMatiere(matiere)
Etudiant -> Formation : contientMatiere(matiere)?
alt matière non trouvée
    Etudiant -> User : throw IllegalArgumentException("La matière n'est pas dans la formation")
else matière trouvée
    Etudiant -> Etudiant : notes = resultats.get(matiere)
    alt notes null ou vides
        Etudiant -> User : throw IllegalArgumentException("Pas de note pour cette matière")
    else notes existantes
        Etudiant -> Etudiant : somme = sum(notes)
        Etudiant -> Etudiant : moyenne = somme / notes.size()
        Etudiant -> User : return moyenne
    end
end
@enduml



Diagramme de séquence Moyenne Generale : 

@startuml
actor "Utilisateur" as User
participant "Etudiant" as Etudiant
participant "Formation" as Formation

User -> Etudiant : calculerMoyenneGenerale()
Etudiant -> Formation : getMatieres()
loop pour chaque matiere
    Etudiant -> Etudiant : notes = resultats.get(matiere)
    alt notes existantes
        Etudiant -> Etudiant : somme = sum(notes)
        Etudiant -> Etudiant : moyenneMatiere = somme / notes.size()
        Etudiant -> Formation : getCoeff(matiere)
        Etudiant -> Etudiant : total += moyenneMatiere * coeff
        Etudiant -> Etudiant : totalCoeff += coeff
    end
end
alt totalCoeff == 0
    Etudiant -> User : throw IllegalArgumentException("Aucune note pour calculer la moyenne générale")
else
    Etudiant -> Etudiant : moyenneGenerale = total / totalCoeff
    Etudiant -> User : return moyenneGenerale
end
@enduml
